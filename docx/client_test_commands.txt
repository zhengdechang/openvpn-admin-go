# 获取客户端列表
curl -X GET http://localhost:8080/api/client/list

# 添加客户端
curl -X POST http://localhost:8080/api/client/add \
  -H "Content-Type: application/json" \
  -d '{"username": "testuser"}'

# 更新客户端
curl -X PUT http://localhost:8080/api/client/update \
  -H "Content-Type: application/json" \
  -d '{"username": "testuser"}'

# 获取客户端配置
curl -X GET http://localhost:8080/api/client/config/testuser

# 吊销客户端证书
curl -X POST http://localhost:8080/api/client/revoke \
  -H "Content-Type: application/json" \
  -d '{"username": "testuser"}'

# 续期客户端证书
curl -X POST http://localhost:8080/api/client/renew \
  -H "Content-Type: application/json" \
  -d '{"username": "testuser"}'

# 删除客户端
curl -X DELETE http://localhost:8080/api/client/delete/testuser
